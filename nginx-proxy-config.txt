1. SSH into your server:
   ssh stavlobel@69.62.119.109

2. Create a backup of current config:
   sudo cp /etc/nginx/sites-available/portfolio /etc/nginx/sites-available/portfolio.backup

3. Edit the nginx configuration:
   sudo nano /etc/nginx/sites-available/portfolio

4. Replace the entire content with this configuration:

----- Copy everything below this line -----
server {
    listen 80;
    listen [::]:80;
    server_name stavlobel.com www.stavlobel.com;

    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name stavlobel.com www.stavlobel.com;

    # SSL Configuration (keep existing SSL settings)
    ssl_certificate /etc/letsencrypt/live/stavlobel.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/stavlobel.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # Proxy to Docker container
    location / {
        proxy_pass http://localhost:3001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Forwarded-Host $host;
        proxy_set_header X-Forwarded-Port $server_port;
        
        # Handle WebSocket connections (if needed in future)
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        
        # Timeouts
        proxy_connect_timeout 30s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;
    }

    # Health check endpoint (optional)
    location /nginx-health {
        access_log off;
        return 200 "healthy\n";
        add_header Content-Type text/plain;
    }
}
----- Copy everything above this line -----

5. Save and exit (Ctrl+X, then Y, then Enter)

6. Test the configuration:
   sudo nginx -t

7. If test passes, reload nginx:
   sudo systemctl reload nginx

8. Test the website:
   curl -I https://stavlobel.com

The website should now proxy to your Docker container!
